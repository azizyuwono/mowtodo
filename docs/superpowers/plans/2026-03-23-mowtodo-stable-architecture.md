// lib/core/database/app_database.dart
import 'package:drift/drift.dart';
import 'package:drift/native.dart';
import 'package:path_provider/path_provider.dart';
import 'package:path/path.dart' as p;
import 'dart:io';
import 'package:sqlite3_flutter_libs/sqlite3_flutter_libs.dart';

part 'app_database.g.dart';

@DataClassName('TodoData')
class Todos extends Table {
  TextColumn get id => text()();
  TextColumn get title => text()();
  TextColumn get description => text().nullable()();
  BoolColumn get isCompleted => boolean().withDefault(const Constant(false))();
  IntColumn get priority => intEnum<Priority>().withDefault(const Constant(1))(); // 0=low, 1=mid, 2=high
  DateTimeColumn get createdAt => dateTime()();
  DateTimeColumn get dueDate => dateTime().nullable()();

  @override
  Set<Column> get primaryKey => {id};
}

enum Priority { low, medium, high }

LazyDatabase _openConnection() {
  return LazyDatabase(() async {
    final dbFolder = await getApplicationDocumentsDirectory();
    final file = File(p.join(dbFolder.path, 'mowtodo.db'));
    if (Platform.isAndroid) {
      await applyCustomSqliteSetup();
    }
    return NativeDatabase(file);
  });
}

@DriftDatabase(tables: [Todos])
class AppDatabase extends _$AppDatabase {
  AppDatabase() : super(_openConnection());

  @override
  int get schemaVersion => 1;

  Future<List<TodoData>> getAllTodos() {
    return select(todos).get();
  }

  Future<List<TodoData>> getActiveTodos() {
    return (select(todos)..where((t) => t.isCompleted.equals(false))).get();
  }

  Future<List<TodoData>> getCompletedTodos() {
    return (select(todos)..where((t) => t.isCompleted.equals(true))).get();
  }

  Future<void> createTodo(TodoData todo) {
    return into(todos).insert(todo);
  }

  Future<bool> updateTodo(TodoData todo) {
    return update(todos).replace(todo);
  }

  Future<bool> deleteTodo(String id) {
    return (delete(todos)..where((t) => t.id.equals(id))).go();
  }

  Future<bool> toggleTodo(String id) async {
    final todo = await (select(todos)..where((t) => t.id.equals(id))).getSingleOrNull();
    if (todo == null) return false;
    return update(todos).replace(todo.copyWith(isCompleted: !todo.isCompleted));
  }

  Future<int> getTodoCount() {
    return countAll().getSingle();
  }
}

// lib/core/extensions/datetime_extensions.dart
extension DateTimeX on DateTime {
  bool get isToday {
    final now = DateTime.now();
    return year == now.year && month == now.month && day == now.day;
  }

  bool get isTomorrow {
    final tomorrow = DateTime.now().add(const Duration(days: 1));
    return year == tomorrow.year && month == tomorrow.month && day == tomorrow.day;
  }

  bool get isOverdue {
    return isBefore(DateTime.now()) && !isToday;
  }

  String get formattedDate {
    if (isToday) return 'Today';
    if (isTomorrow) return 'Tomorrow';
    return '$day/$month/$year';
  }
}

import 'package:uuid/uuid.dart';

enum Priority {
  low,
  medium,
  high,
}

class Todo {
  final String id;
  final String title;
  final String? description;
  final bool isCompleted;
  final Priority priority;
  final DateTime createdAt;
  final DateTime? dueDate;

  Todo({
    String? id,
    required this.title,
    this.description,
    this.isCompleted = false,
    this.priority = Priority.medium,
    DateTime? createdAt,
    this.dueDate,
  })  : id = id ?? const Uuid().v4(),
        createdAt = createdAt ?? DateTime.now();

  Todo copyWith({
    String? id,
    String? title,
    String? description,
    bool? isCompleted,
    Priority? priority,
    DateTime? createdAt,
    DateTime? dueDate,
  }) {
    return Todo(
      id: id ?? this.id,
      title: title ?? this.title,
      description: description ?? this.description,
      isCompleted: isCompleted ?? this.isCompleted,
      priority: priority ?? this.priority,
      createdAt: createdAt ?? this.createdAt,
      dueDate: dueDate ?? this.dueDate,
    );
  }

  @override
  bool operator ==(Object other) =>
      identical(this, other) ||
      other is Todo &&
          runtimeType == other.runtimeType &&
          id == other.id;

  @override
  int get hashCode => id.hashCode;

  @override
  String toString() => 'Todo(id: $id, title: $title, isCompleted: $isCompleted)';
}

// lib/repositories/todo_repository.dart
import '../core/database/app_database.dart';
import '../models/todo.dart';

class TodoRepository {
  final AppDatabase _database;

  TodoRepository(this._database);

  Future<List<Todo>> getAllTodos() async {
    final todos = await _database.getAllTodos();
    return todos.map(_toDomain).toList();
  }

  Future<List<Todo>> getActiveTodos() async {
    final todos = await _database.getActiveTodos();
    return todos.map(_toDomain).toList();
  }

  Future<List<Todo>> getCompletedTodos() async {
    final todos = await _database.getCompletedTodos();
    return todos.map(_toDomain).toList();
  }

  Future<void> createTodo(Todo todo) async {
    await _database.createTodo(_toDatabase(todo));
  }

  Future<bool> updateTodo(Todo todo) async {
    return _database.updateTodo(_toDatabase(todo));
  }

  Future<bool> deleteTodo(String id) async {
    return _database.deleteTodo(id);
  }

  Future<bool> toggleTodo(String id) async {
    return _database.toggleTodo(id);
  }

  Future<int> getTodoCount() async {
    return _database.getTodoCount();
  }

  // Mappers
  Todo _toDomain(TodoData data) {
    return Todo(
      id: data.id,
      title: data.title,
      description: data.description,
      isCompleted: data.isCompleted,
      priority: Priority.values[data.priority],
      createdAt: data.createdAt,
      dueDate: data.dueDate,
    );
  }

  TodoData _toDatabase(Todo todo) {
    return TodoData(
      id: todo.id,
      title: todo.title,
      description: todo.description,
      isCompleted: todo.isCompleted,
      priority: todo.priority.index,
      createdAt: todo.createdAt,
      dueDate: todo.dueDate,
    );
  }

// lib/providers/todo_provider.dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../core/database/app_database.dart';
import '../models/todo.dart';
import '../repositories/todo_repository.dart';
import 'database_provider.dart';

final todoRepositoryProvider = Provider((ref) {
  final db = ref.watch(databaseProvider).valueOrNull;
  if (db == null) throw StateError('Database not initialized');
  return TodoRepository(db);
});

final todoListProvider = FutureProvider<List<Todo>>((ref) async {
  final repository = ref.watch(todoRepositoryProvider);
  return repository.getAllTodos();
});

final activeTodoCountProvider = FutureProvider<int>((ref) async {
  final todos = await ref.watch(todoListProvider.future);
  return todos.where((t) => !t.isCompleted).length;
});

final completedTodoCountProvider = FutureProvider<int>((ref) async {
  final todos = await ref.watch(todoListProvider.future);
  return todos.where((t) => t.isCompleted).length;
});

final todoNotifierProvider = StateNotifierProvider<TodoNotifier, AsyncValue<List<Todo>>>((ref) {
  final repository = ref.watch(todoRepositoryProvider);
  return TodoNotifier(repository);
});

class TodoNotifier extends StateNotifier<AsyncValue<List<Todo>>> {
  final TodoRepository _repository;

  TodoNotifier(this._repository) : super(const AsyncValue.loading()) {
    _init();
  }

  Future<void> _init() async {
    try {
      final todos = await _repository.getAllTodos();
      state = AsyncValue.data(todos);
    } catch (e) {
      state = AsyncValue.error(e, StackTrace.current);
    }
  }

  Future<void> addTodo(String title, {String? description, Priority priority = Priority.medium}) async {
    final todo = Todo(title: title, description: description, priority: priority);
    try {
      await _repository.createTodo(todo);
      await refresh();
    } catch (e) {
      state = AsyncValue.error(e, StackTrace.current);
    }
  }

  Future<void> toggleTodo(String id) async {
    try {
      await _repository.toggleTodo(id);
      await refresh();
    } catch (e) {
      state = AsyncValue.error(e, StackTrace.current);
    }
  }

  Future<void> deleteTodo(String id) async {
    try {
      await _repository.deleteTodo(id);
      await refresh();
    } catch (e) {
      state = AsyncValue.error(e, StackTrace.current);
    }
  }

  Future<void> refresh() async {
    state = const AsyncValue.loading();
    try {
      final todos = await _repository.getAllTodos();
      state = AsyncValue.data(todos);
    } catch (e) {
      state = AsyncValue.error(e, StackTrace.current);
    }
  }
}

import 'package:flutter_riverpod/flutter_riverpod.dart';

class HomeScreen extends ConsumerWidget {
  const HomeScreen({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final todosAsync = ref.watch(todoNotifierProvider);
    final activeCount = ref.watch(activeTodoCountProvider);
    final completedCount = ref.watch(completedTodoCountProvider);

    return Scaffold(
      appBar: AppBar(title: const Text('MowTodo')),
      body: Column(
        children: [
          // Stats
          todosAsync.when(
            data: (todos) => TodoStats(
              total: todos.length,
              active: todos.where((t) => !t.isCompleted).length,
              completed: todos.where((t) => t.isCompleted).length,
            ),
            loading: () => const SizedBox(height: 80),
            error: (err, st) => const SizedBox(height: 80),
          ),

          // Add input
          AddTodoInput(
            onAdd: (title, {description, priority}) {
              ref.read(todoNotifierProvider.notifier).addTodo(
                    title,
                    description: description,
                    priority: priority,
                  );
            },
          ),

          // Todo list
          Expanded(
            child: todosAsync.when(
              data: (todos) => todos.isEmpty
                  ? const EmptyState()
                  : ListView.builder(
                      itemCount: todos.length,
                      itemBuilder: (context, index) {
                        final todo = todos[index];

                          },
                          onDelete: () {
                            ref.read(todoNotifierProvider.notifier).deleteTodo(todo.id);
                          },
                        );
                      },
                    ),
              loading: () => const Center(
                child: CircularProgressIndicator(),
              ),
              error: (err, st) => Center(

// test/integration/persistence_test.dart
import 'package:flutter_test/flutter_test.dart';
import 'package:mowtodo/core/database/app_database.dart';
import 'package:mowtodo/models/todo.dart';
import 'package:mowtodo/repositories/todo_repository.dart';

void main() {
  group('Persistence Integration', () {
    late AppDatabase database;
    late TodoRepository repository;

    setUpAll(() async {
      // Use in-memory database for testing
      database = AppDatabase();
      repository = TodoRepository(database);
    });

    tearDownAll(() async {
      await database.close();
    });

    test('create and retrieve todo', () async {
      final todo = Todo(title: 'Integration Test');
      await repository.createTodo(todo);

      final todos = await repository.getAllTodos();
      expect(todos, isNotEmpty);
      expect(todos.first.title, 'Integration Test');
    });

    test('toggle todo updates database', () async {
      final todo = Todo(id: 'toggle-test', title: 'Toggle Me');
      await repository.createTodo(todo);

      await repository.toggleTodo('toggle-test');

      final todos = await repository.getAllTodos();
      final updated = todos.firstWhere((t) => t.id == 'toggle-test');
      expect(updated.isCompleted, true);
    });

    test('delete todo removes from database', () async {
      final todo = Todo(id: 'delete-test', title: 'Delete Me');
      await repository.createTodo(todo);

      int countBefore = (await repository.getAllTodos()).length;
      await repository.deleteTodo('delete-test');
      int countAfter = (await repository.getAllTodos()).length;

      expect(countAfter, lessThan(countBefore));
    });
  });
}
